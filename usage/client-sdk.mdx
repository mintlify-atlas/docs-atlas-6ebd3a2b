## Installation

Install the PyWorker repository and dependencies:

```bash
git clone https://github.com/ByQwank/pyworker
cd pyworker
pip install uv
uv venv -p 3.12
source .venv/bin/activate
uv pip install -r requirements.txt
```

The client requires the Vast Serverless SDK:

```python
from vastai import Serverless
```

## Authentication

Set your Vast API key as an environment variable:

```bash
export VAST_API_KEY=<your_api_key>
```

The SDK will automatically use this key for authentication.

## Basic Usage

### Initialize the Client

Create an async context with the Serverless client:

```python
import asyncio
from vastai import Serverless

async def main():
    async with Serverless() as client:
        # Your code here
        pass

asyncio.run(main())
```

### Get an Endpoint

Retrieve a configured endpoint by name:

```python
endpoint_name = "my-comfyui-endpoint"
endpoint = await client.get_endpoint(name=endpoint_name)
```

### Making Requests

All requests use the `/generate/sync` route with a cost parameter:

```python
payload = {
    "input": {
        "request_id": str(uuid.uuid4()),
        # ... additional fields
    }
}

response = await endpoint.request("/generate/sync", payload, cost=100)
```

The `cost` parameter represents the workload cost for routing purposes.

## Text-to-Image Generation

Generate images using the Text2Image modifier:

```python
import uuid
from vastai import Serverless

async def generate_image(
    client: Serverless,
    endpoint_name: str,
    prompt: str,
    width: int = 512,
    height: int = 512,
    steps: int = 20,
    seed: int = None
):
    endpoint = await client.get_endpoint(name=endpoint_name)
    
    payload = {
        "input": {
            "request_id": str(uuid.uuid4()),
            "modifier": "Text2Image",
            "modifications": {
                "prompt": prompt,
                "width": width,
                "height": height,
                "steps": steps,
                "seed": seed,
            },
        }
    }
    
    return await endpoint.request("/generate/sync", payload, cost=100)
```

## Handling Responses

### Extract Image Filename

Parse the ComfyUI response to get the generated image filename:

```python
def extract_filename(response: dict) -> str | None:
    """Extract the generated image filename from ComfyUI response"""
    if "comfyui_response" in response:
        for data in response["comfyui_response"].values():
            if isinstance(data, dict) and "outputs" in data:
                for node_output in data["outputs"].values():
                    if "images" in node_output and node_output["images"]:
                        return node_output["images"][0].get("filename")
    return None
```

### Download Generated Images

Fetch images from the worker's `/view` endpoint:

```python
import aiohttp
import os

async def download_image(
    worker_url: str,
    filename: str,
    local_path: str
) -> bool:
    """Download image from worker to local filesystem"""
    url = f"{worker_url}/view"
    params = {"filename": filename, "type": "output"}
    
    async with aiohttp.ClientSession() as session:
        async with session.get(url, params=params, ssl=False) as resp:
            if resp.status == 200:
                os.makedirs(os.path.dirname(local_path), exist_ok=True)
                image_data = await resp.read()
                with open(local_path, "wb") as f:
                    f.write(image_data)
                return True
    return False

# Usage:
worker_url = response.get("url", "")
if "response" in response:
    filename = extract_filename(response["response"])
    if filename:
        await download_image(worker_url, filename, "output.png")
```

## S3 Upload (Optional)

Upload generated images to S3-compatible storage:

### Configure S3 Credentials

```bash
export S3_ENDPOINT_URL="https://your-account.r2.cloudflarestorage.com"
export S3_BUCKET_NAME="my-bucket"
export S3_ACCESS_KEY_ID="your-access-key-id"
export S3_SECRET_ACCESS_KEY="your-secret-access-key"
```

### Upload to S3

```python
import boto3
from botocore.config import Config

def get_s3_client():
    return boto3.client(
        "s3",
        endpoint_url=os.getenv("S3_ENDPOINT_URL"),
        aws_access_key_id=os.getenv("S3_ACCESS_KEY_ID"),
        aws_secret_access_key=os.getenv("S3_SECRET_ACCESS_KEY"),
        config=Config(signature_version="s3v4"),
    )

def upload_to_s3(local_path: str, s3_key: str, bucket: str):
    s3_client = get_s3_client()
    s3_client.upload_file(
        local_path,
        bucket,
        s3_key,
        ExtraArgs={"ContentType": "image/png"}
    )
    return f"{os.getenv('S3_ENDPOINT_URL')}/{bucket}/{s3_key}"
```

Requires `boto3`: `pip install boto3`

## CLI Client

Run the included client from the command line:

```bash
# Default prompt
python -m workers.comfyui-json.client

# Custom prompt
python -m workers.comfyui-json.client --prompt "a cat sitting on a rainbow"

# With options
python -m workers.comfyui-json.client \
  --prompt "sunset" \
  --width 1024 \
  --height 1024 \
  --steps 30

# Custom workflow file
python -m workers.comfyui-json.client --workflow my_workflow.json

# Enable S3 upload
python -m workers.comfyui-json.client --s3
```

### CLI Options

| Flag | Default | Description |
|------|---------|-------------|
| `--endpoint` | `my-comfyui-endpoint` | Vast endpoint name |
| `--prompt` | (default) | Text prompt for image generation |
| `--workflow` | (none) | Path to custom workflow JSON file |
| `--width` | 512 | Image width in pixels |
| `--height` | 512 | Image height in pixels |
| `--steps` | 20 | Number of denoising steps |
| `--seed` | (random) | Random seed for reproducibility |
| `--s3` | (disabled) | Upload generated images to S3 |

## Complete Example

Here's a full example from `client.py:173-220`:

```python
import asyncio
import random
import uuid
from vastai import Serverless

async def demo_prompt(
    client: Serverless,
    endpoint_name: str,
    prompt: str,
    width: int,
    height: int,
    steps: int,
    seed: int | None,
):
    if seed is None:
        seed = random.randint(0, 2**32 - 1)

    print(f"Generating: {prompt}")
    print(f"Size: {width}x{height}, Steps: {steps}, Seed: {seed}")

    # Get endpoint
    endpoint = await client.get_endpoint(name=endpoint_name)
    
    # Prepare payload
    payload = {
        "input": {
            "request_id": str(uuid.uuid4()),
            "modifier": "Text2Image",
            "modifications": {
                "prompt": prompt,
                "width": width,
                "height": height,
                "steps": steps,
                "seed": seed,
            },
        }
    }
    
    # Make request
    response = await endpoint.request("/generate/sync", payload, cost=100)
    
    print("Generation complete!")
    print(f"Worker URL: {response.get('url', '')}")
    
    # Extract and download image
    if "response" in response:
        filename = extract_filename(response["response"])
        if filename:
            worker_url = response.get("url", "")
            await download_image(worker_url, filename, f"comfy_{seed}.png")

# Run
async def main():
    async with Serverless() as client:
        await demo_prompt(
            client,
            endpoint_name="my-comfyui-endpoint",
            prompt="a beautiful sunset over mountains",
            width=512,
            height=512,
            steps=20,
            seed=None
        )

asyncio.run(main())
```

## Error Handling

Handle authentication errors:

```python
try:
    async with Serverless() as client:
        # Your code
        pass
except AttributeError as e:
    if "API key" in str(e):
        print("Error: VAST_API_KEY environment variable not set")
    else:
        print(f"Error: {e}")
except Exception as e:
    print(f"Error: {e}")
```
